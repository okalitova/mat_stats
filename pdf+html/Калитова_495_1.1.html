colors = ['red', 'blue', 'black', 'purple']
for theta in [3., 100., 1000.]:
    samples = sts.uniform.rvs(size=10000, loc=0, scale=theta)
    N = np.size(samples)
    res = [[] for i in range(4)]
    for n in range(1, N + 1) :
        res[0].append(2. * np.mean(samples[:n]))
        res[1].append(np.mean(samples[:n]) + np.max(samples[:n]) / 2.)
        res[2].append(np.min(samples[:n]) + np.max(samples[:n]))
        res[3].append((n + 1.) / n * np.max(samples[:n]))
    for i in range(4):
        res[i] = np.abs(np.array(res[i]) - theta)
    x = np.arange(N) + 1;
    plt.ylim([0., theta / 50])
    plt.xlim([1., N])
    plt.title("$\\theta = " + str(theta) + "$")
    plt.xlabel("# samples")
    plt.ylabel("statistics")
    for i in range(4):
        plt.plot(x, res[i], color=colors[i])
    plt.show()

После проведения еще двух экспериментов подтверждается, что зеленая((n + 1) X(1)) - самая плохая оценка. Красная(2 Х) и синяя(X + X(n) / 2) явно проигрывают черной $(X_{(n+1)} + X_{(1)})$ и фиолетовой((n+1)/n X(n)). Но при этом так же становится видно, что синяя() лучше красной(), а фиолетовая((n+1)/n X(n)) лучше черной(X(n+1) + X(1)). То есть самая лучшая оценка фиолетовая((n+1)/n * X(n)).
